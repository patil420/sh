// Program to store roll numbers of students and search using Linear & Binary Search

#include <stdio.h>

// Function for Linear Search
void linearSearch(int arr[], int n, int key) {
    for(int i = 0; i < n; i++) {
        if(arr[i] == key) {
            printf("Student attended (found by Linear Search)\n");
            return;
        }
    }
    printf("Student not found (Linear Search)\n");
}

// Function for Binary Search (Array must be sorted)
void binarySearch(int arr[], int n, int key) {
    int low = 0, high = n - 1, mid;
    while(low <= high) {
        mid = (low + high) / 2;
        if(arr[mid] == key) {
            printf("Student attended (found by Binary Search)\n");
            return;
        } else if(arr[mid] < key)
            low = mid + 1;
        else
            high = mid - 1;
    }
    printf("Student not found (Binary Search)\n");
}

int main() {
    int n, key;

    // Input number of students
    printf("Enter number of students: ");
    scanf("%d", &n);

    int arr[n];

    // Input roll numbers in random order
    printf("Enter roll numbers:\n");
    for(int i = 0; i < n; i++)
        scanf("%d", &arr[i]);

    // Roll number to search
    printf("Enter roll number to search: ");
    scanf("%d", &key);

    // Perform Linear Search
    linearSearch(arr, n, key);

    // Note: For Binary Search, array must be sorted
    // Simple Bubble Sort for demonstration
    for(int i = 0; i < n - 1; i++)
        for(int j = 0; j < n - i - 1; j++)
            if(arr[j] > arr[j + 1]) {
                int temp = arr[j];
                arr[j] = arr[j + 1];
                arr[j + 1] = temp;
            }

    // Perform Binary Search
    binarySearch(arr, n, key);

    return 0;
}
